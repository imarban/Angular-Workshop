Directives

Declare directives by:
< directive >            //Element
< div directive>         //Atribute
< div class="directive"> //Class

            angular
                .module(app_name)
                .directive(directive_name, directive_name);

            directive_name.$inject = [dependencies];

            function directive_name(dependencies) {
                return {
                    controller: 'nameCtrl'
                    require: '^', //requires other directives Ctrl
                    scope: false, //default receives sames scope as its parent
                    scope: true, //creates own scope has access to parent
                    scope : {     //Isolate scope
                        two-way: '=',
                        meth-bind: '&',
                        read-only: '@'
                    }
                    link: function(scope, element, attr, ctrl) {

                    }
                }

            }
